Module de scripts (à exécuter localement) spécifiques au dépôt d'exercices et complétant les commandes définies dans le manifeste..

Modifié le 09/02/23 @author: remy

La fonction exec() exécute les tâches spécifiques complémentaires. Elle est appelée lors de l'instanciation de la classe Execlocal.

 24def exec(data):
 25    """
 26    Forme les exos individuels et les feuilles d'exos LateX.
 27
 28    - Lit les thèmes depuis le fichier `_codes.csv`
 29    - Pour chaque thème, forme le fichier LateX `A_`
 30    - Pour chaque exercice, forme le fichier LateX `Aexo_`
 31    - Renvoie un journal de l'exécution
 32
 33
 34    Returns
 35    -------
 36    log : TYPE str
 37        DESCRIPTION journal de l'exécution des scripts.
 38
 39    """
 40    log = lineprefix + 'Scripts du module spécifique (fonction exec())'
 41
 42    # Feuilles par thèmes
 43    theme = getThemes()
 44    log += '\n \t\t\t Création modification des feuilles LateX'
 45    for code in theme:
 46        if aecrirefeuilleA(code):
 47            log += ecritFeuilleA(code, theme[code])
 48
 49    # Exercices individuels "Aexo_"
 50    log += '\n\n \t \t \t Création modification des "Aexo_" Latex individuels'
 51    lili = listeExos()
 52    for theme in lili:
 53        for nomexo in lili[theme]:
 54            if aecrireAexo(nomexo):
 55                log += ecritAexo(nomexo)
 56
 57    return {'log': log , 'specific_results': {} }

 60def ecritFeuilleA(code, dscrpt):
 61    """
 62    Exécute les scripts Python spécifiques à ce dépôt.
 63
 64     ici
 65        - création ou maj feuille d'exos Latex d'un thème
 66
 67
 68    Parameters
 69    ----------
 70    code : TYPE str
 71        DESCRIPTION code du thème dans le fichier `_code.csv`.
 72    dscrpt : TYPE str
 73        DESCRIPTION description du thème dans le fichier `_code.csv`.
 74
 75    Returns
 76    -------
 77    log : TYPE str
 78        DESCRIPTION journal de l'exécution.
 79
 80    """
 81    log = '\n \t \t \t' + code + ' : ' + dscrpt
 82    exos_list = glob.glob(relpath + 'E' + code + '*.tex')
 83    exos_list.sort()
 84    dates = map(os.path.getmtime, exos_list)
 85    date_inputs = max(dates)
 86
 87    # début fichier
 88    Latex = '\\input{exopdf} \n'
 89    Latex += '\\begin{document} \n'
 90
 91    # énoncés
 92    Latex += '\\chead{ ' + dscrpt + ': énoncés.}\n'
 93    Latex += '\\begin{enumerate}\n'
 94    for exo in exos_list:
 95        exo = os.path.basename(exo)
 96        Latex += '  \\item \\input{' + exo + '} \n'
 97    Latex += '\\end{enumerate} \n'
 98
 99    # corrigés
100    Latex += '\\clearpage \n'
101    Latex += '\\chead{' + dscrpt + ': corrigés.}\n'
102    Latex += '\\begin{enumerate}\n'
103    for exo in exos_list:
104        exoC = exo.replace('E' + code, 'C' + code)
105        if os.path.isfile(exoC):
106            date_inputs = max(date_inputs, os.path.getmtime(exoC))
107            Latex += '  \\item \\input{' + os.path.basename(exoC) + '} \n'
108        else:
109            Latex += '  \\item ' + os.path.basename(exoC) + ' manque. \n'
110    Latex += '\\end{enumerate} \n'
111
112    # fin fichier
113    Latex += '\\end{document}'
114
115    # écrit le Latex dans un fichier A#
116    pathA = relpath + 'A_' + code + '.tex'
117
118    with open(pathA, 'w') as fifi:
119        fifi.write(Latex)
120        log += ' NOUVEAU OU MODIFIÉ ' + pathA
121
122    return log

125def aecrirefeuilleA(code):
126    """
127    Renvoie VRAI ou FAUX.
128
129    VRAI si le fichier LateX "A" du thème doit être créé ou modifié
130    c'est à dire si
131        - le fichier "A" n'existe pas
132        ou
133        - la liste des fichiers "E" ou "C" n'est pas dans celle des input
134
135    Parameters
136    ----------
137    code : TYPE str
138        DESCRIPTION code du theme.
139
140    Returns
141    -------
142    bool.
143
144    """
145    pathA = relpath + 'A_' + code + '.tex'
146
147    bouboule = False
148    if not os.path.isfile(pathA):
149        bouboule = True
150    else:
151        inputs = set(scantex.get_liste_inputs(pathA))
152        fifi = set(listeExos()[code]) | set(listeCorriges()[code])
153        bouboule = not (fifi <= inputs)
154    return bouboule

157def aecrireAexo(nomexo):
158    """
159    Renvoie VRAI si le fichier "Aexo" de l'exercice doit être créé
160    ou modifié. C'est à dire si :
161        - le fichier "Aexo_" n'existe pas'
162        ou
163        - le fichier "Aexo_" ne contient pas de corrigé alors qu'il existe
164        un fichier "C".
165    Parameters
166    ----------
167    nomexo : TYPE str
168        DESCRIPTION nom du fichier "E" d'un exercice.
169
170    Returns
171    -------
172    TYPE bool
173
174    """
175    exospath = dp_data['relative_path']
176    nomexoC = nomexo.replace('E', 'C')
177    nomexoA = nomexo.replace('E', 'Aexo_')
178    pathC = exospath + nomexoC
179    pathA = exospath + nomexoA
180    bouboule = False
181    if not os.path.isfile(pathA):
182        bouboule = True
183    elif os.path.isfile(pathC):
184        with open(pathA) as fifi:
185            latex = fifi.read()
186            if "pas encore de corrigé" in latex:
187                bouboule = True
188    return bouboule

191def ecritAexo(nomexo):
192    """
193    Forme le fichier latex "Aexo_" d'un exercice particulier'
194    et l'écrit dans le dépôt.
195
196
197    Parameters
198    ----------
199    nomexo :
200        TYPE str
201        DESCRIPTION  nom du fichier de l'exercice (énoncé).
202
203    Returns
204    -------
205    None ou le nom du fichier Aexo modifié.
206
207    """
208    log = ''
209    exospath = dp_data['relative_path']
210    nomexoE = nomexo
211    nomexoC = nomexo.replace('E', 'C')
212    nomexoA = nomexo.replace('E', 'Aexo_')
213    pathE = exospath + nomexoE
214    pathC = exospath + nomexoC
215    pathA = exospath + nomexoA
216
217    # début fichier
218    Latex = '\\input{exo4ht} \n'
219    Latex += '\\begin{document} \n'
220    Latex += 'Énoncé \n\n'
221    Latex += '\\input{' + os.path.basename(pathE) + '}\n\n'
222    Latex += 'Corrigé \n\n'
223    if os.path.isfile(pathC):
224        Latex += '\\input{' + os.path.basename(pathC) + '}\n'
225    else:
226        Latex += 'pas encore de corrigé\n'
227    Latex += '\\end{document}'
228    # print(Latex)
229
230    """
231    On écrit toujours.
232    """
233    with open(pathA, 'w') as fifi:
234        fifi.write(Latex)
235        log += '\n \t\t\t' + os.path.basename(pathA)
236    return log
237    # sinon ne renvoie rien

278def getThemes():
279    """
280    Lit les thèmes depuis le fichier _codes.csv
281
282    Returns
283    -------
284    dictThemes : TYPE dict
285        DESCRIPTION key=code, val=description.
286
287    """
288    dictThemes = {}
289    with open('../math-exos/' + '_codes.csv', newline='') as csvfile:
290        riri = csv.reader(csvfile, delimiter=';')
291        for row in riri:
292            dictThemes[row[0]] = row[1]
293        dictThemes.pop('codetheme')
294    return dictThemes

297def listeExos():
298    """
299
300    Returns
301    -------
302    dictionnaire
303        - clé = code d'un thème
304        - valeur = liste des noms des fichiers d'énoncé
305
306    """
307    themes = getThemes()
308    exospath = relpath
309    files = os.listdir(exospath)
310
311    exercices = {}
312
313    for code in themes:
314        exercices[code] = []
315        regexpr = "^E" + code + "[0-9]{2}.tex"
316        for file in files:
317            if re.match(regexpr, file):
318                exercices[code].append(file)
319        exercices[code].sort()
320    return exercices
